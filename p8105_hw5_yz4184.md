fill_iris_missing = function(vector){
  if(is.numeric(vector)){
    vector= replace_na(vector, mean(vector, na.rm = TRUE))
  }
  if(is.character(vector)){
    vector= replace_na(vector, "virginica")
  }
  return(vector)
}
